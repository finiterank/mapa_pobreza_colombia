colmap@data$id <- rownames(colmap@data)
departamentos.df <- fortify(colmap)

## Regions defined for each Polygons

departamentos.df <- join(departamentos.df, colmap@data, by = "id")
# departamentos.df$NAME_1 <- mapvalues(departamentos.df$NAME_1,
# from=c('Atl\xe1ntico', 'Choc\xf3', 'C\xf3rdoba', 'Vaup\xe9s',
# 'Bol\xedvar', 'San Andr\xe9s y Providencia', 'Boyac\xe1',

# Providencia', 'Boyacá', 'Caquetá', 'Guainía','Nariño', 'Quindío'))
pobreza_depto <- ddply(pobreza, ~Departamento, summarise, mean = mean(Incidencia))
pobreza_depto$Departamento <- mapvalues(pobreza_depto$Departamento, from = c("C/marca.", 
    "Norte de Sant.", "San Andrés, Prov. y Sta Cat.", "Valle", "Guajira"), 
    to = c("Cundinamarca", "Norte de Santander", "San Andrés y Providencia", 
        "Valle del Cauca", "La Guajira"))
pobreza_depto <- pobreza_depto[-5, ]

departamentos.df <- merge(departamentos.df, pobreza_depto, by.x = "NAME_1", 
    by.y = "Departamento", all.x = T, all.y = F)


mapa_pobreza <- ggplot(data = departamentos.df, aes(x = long, y = lat, group = group)) + 
    th
mapa_pobreza <- mapa_pobreza + geom_polygon(aes(fill = mean))
mapa_pobreza <- mapa_pobreza + geom_path(color = "white", linestyle = 2, aes(group = group))
mapa_pobreza <- mapa_pobreza + coord_equal()
mapa_pobreza <- mapa_pobreza + scale_fill_gradient(name = "Promedio de incidencia\nde pobreza por\ndepartamento", 
    low = "#ffffcc", high = "#ff4444", space = "Lab", na.value = "grey50", guide = "colourbar")
mapa_pobreza <- mapa_pobreza + labs(title = "Pobreza por departamento")
